Water status

{{
    local feat = view.feature_at(0, 0):gsub("_", " ")
    if you.flying() then
        if feat == "deep water" then
            return "Du fliegst über tiefes Wasser.
        elseif feat == "shallow water" then
            return "Du fliegst über seichtes Wasser.
        else
            return "Du fliegst über Wasser.
        end
    else
        local verb = you.race() == "Merfolk" and "schwimmst in" or "stehst in"
        if feat == "deep water" then
            return "Du " .. verb .. " tiefem Wasser.
        elseif feat == "shallow water" then
            return "Du " .. verb .. " seichtem Wasser.
        else
            return "Du " .. verb .. " Wasser.
        end
    end
}}

Zot status

auszehren.
