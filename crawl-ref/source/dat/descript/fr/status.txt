    local feat = view.feature_at(0, 0):gsub("_", " ")
    if you.flying() then
        if feat == "deep water" then
            return "Vous volez au-dessus d'eaux profondes.
        elseif feat == "shallow water" then
            return "Vous volez au-dessus d'eaux peu profondes.
        else
            return "Vous volez au-dessus de l'eau.
        end
    else
        local verb = you.race() == "Merfolk" and "nage dans" or "trouve dans"
        if feat == "deep water" then
            return "Ihr " .. verb .. " eau profonde.
        elseif feat == "shallow water" then
            return "Ihr " .. verb .. " eaux peu profondes.
        else
            return "Ihr " .. verb .. " Eau.
        end
    end
}}
