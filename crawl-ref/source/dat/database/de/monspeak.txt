############################################################
# Says/shouts/etc. to/at @foe@.
# The " to/at @foe@" bit is dropped if foe is player.
# Note that @speaks_to_foe@ is set by DCSS using a monster-specific
# verb (shouts, screams, etc.), but it follows the same rule
# of dropping the " to/at @foe@" bit if foe is player.
############################################################
%%%%
_to_foe_

[@foe@==you?: zu {dat}@foe@]
%%%%
_at_foe_

[@foe@==you?: {acc}@foe@ an]
%%%%
_asks_foe_

[@foe@==you?fragt:fragt {acc}@foe@]
%%%%
_says_to_foe_

[@foe@==you?sagt:sagt zu {dat}@foe@]
%%%%
_shouts_at_foe_

[@foe@==you?schreit:schreit {acc}@foe@ an]
%%%%
_screams_at_foe_

[@foe@==you?schreit:schreit {acc}@foe@ an]
%%%%
